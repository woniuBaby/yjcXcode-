```
在命令行中输入一下命令，添加临时环境量
export http_proxy=socks5://127.0.0.1:1080
export https_proxy=socks5://127.0.0.1:1080

设置git环境变量
git config –global http.proxy 'socks5://127.0.0.1:1080'

//全局统一代理
export ALL_PROXY=http://127.0.0.1:1087

//查看
export -p
```
